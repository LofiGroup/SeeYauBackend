Install libraries from requirements file:
pip install -r /path/to/requirements.txt

Write dependencies to requirements file:
pip freeze > requirements.txt

Delete remote git branch:
git push origin --delete remote_branch

Set git default line-endings format to LF
git config --global core.autocrlf input

Git clone using ssh:
git clone git@github.com:repo_path

Mysql shell:
mysql -u sql_user -p

# $something$ means that
Mysql use db:
use $db_name$;

Mysql update all column in table:
update $table_name$ set $column_name$ = $value$;
